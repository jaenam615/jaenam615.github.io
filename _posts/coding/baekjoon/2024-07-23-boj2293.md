## 코드
```python
import sys
input = sys.stdin.readline

def dp(k, coins):
    cache = [0] * (k+1)
    cache[0] = 1

    for coin in coins:
        for i in range(coin, k+1):
            cache[i] += cache[i-coin]
    
    return cache[k]

n, k = map(int, input().split())
coins = []

for _ in range(n):
    coins.append(int(input()))

coins.sort()

print(dp(k, coins))
```
